## Basic code

```
const MainModalExample = () => {
  const [open, setOpen] = useState(false);

  const handleOpen = () => {
    setOpen(true);
  };

  const handleClose = () => {
    setOpen(false);
  };

  return (
    <Main>
      <button type="button" onClick={handleOpen}>Open modal</button>
      <MainModal open={open} onClose={handleClose}>
        <MainModal.Title>
          Title
        </MainModal.Title>
        <MainModal.Content>
          Content
        </MainModal.Content>
      </MainModal>
    </Main>
  );
};
```
