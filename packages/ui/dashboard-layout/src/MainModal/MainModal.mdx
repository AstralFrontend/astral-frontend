# MainModal

<Props of={MainModal} />

**Для того, чтобы модалка была не на весь экран, а была в контейнере, прокинуть в container прокинуть ref контейнера. Необходимо, чтобы сама модалка тоже находилась в контейнере**

## Basic code

```
const MainModalExample = ({ size }) => {
  const [open, setOpen] = useState(false);
  const containerRef = useRef();

  const handleOpen = () => {
    setOpen(true);
  };

  const handleClose = () => {
    setOpen(false);
  };

  return (
    <div ref={containerRef}>
      <button type="button" onClick={handleOpen}>Open modal</button>
      <ThemeProvider theme={new AstralEDOTheme()}>
        <MainModal open={open} size={size} container={containerRef.current} onClose={handleClose}>
          <MainModalTitle>
              Title
          </MainModalTitle>
          <MainModalContent>
              Content
          </MainModalContent>
        </MainModal>
      </ThemeProvider>
    </div>
  );
};
```
