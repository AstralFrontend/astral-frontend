# Checkbox

<Props of={Checkbox} />

## Props
  - disabled: bool
  - checked: bool
  - labelPlacement: oneOf(['end', 'start', 'top', 'bottom']),
  - error: oneOfType([PropTypes.bool, PropTypes.string])
  - className: string
  - helperText: string,
  - label: node
  - onBlur: func
  - onFocus: func
  - value: any
  - onChange: func.isRequired

default:
  - labelPlacement = 'end'

## Example with label

```
const ExampleCheckbox = () => {
  const checkboxValue = 'check A value';
  const [currentValue, setCurrentValue] = useState(null);

  console.log('check A value', currentValue);

  return (
    <Checkbox
      checked={currentValue === checkboxValue}
      label="label"
      value={checkboxValue}
      onChange={(event) => {
        const { target } = event;
        const { checked, value } = target;

        if (checked) setCurrentValue(value);
        if (!checked) setCurrentValue(null);
      }}
    />
  );
};
```
